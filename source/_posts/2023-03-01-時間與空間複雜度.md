### [LIOJ | 簡易排序](https://oj.lidemy.com/problem/1035)

使用到 `.sort`內建函數

簡單來說`.sort`可以傳入一個 function 並依照 return 的值對**原本**陣列進行排序

簡易記法，假設傳入的值為 a,b 分別是 3 與 5

如果 return 值為 -2 (a-b) 則陣列保持不變(由小到大)

如果 return 值 2 (b-a) 則a,b對調，且陣列排序(由大到小)

```js
var readline = require('readline');
var rl = readline.createInterface({
  input: process.stdin
});

var lines = []

// 讀取到一行，先把這一行加進去 lines 陣列，最後再一起處理
rl.on('line', function (line) {
  lines.push(line)
});

// 輸入結束，開始針對 lines 做處理
rl.on('close', function() {
  solve(lines)
})

// 上面都不用管，只需要完成這個 function 就好，可以透過 lines[i] 拿取內容
function solve(lines) {
 let arr=[]
  for(let i=1;i<lines.length;i++){
    arr.push(lines[i])
  }
  //3,5
  arr.sort((a,b)=>{
    return a-b
  })
  for(let i=0;i<arr.length;i++){
    console.log(arr[i])
  }

}

```

使用`counting sort`排序方式 

原理：因為數入數值是固定的，所以先準備好特定範圍內的容器，然後分別放入

最後再用迴圈讀取每個容器數值為多少就是需要重複印出幾次

```js
var readline = require('readline');
var rl = readline.createInterface({
  input: process.stdin
});

var lines = []

// 讀取到一行，先把這一行加進去 lines 陣列，最後再一起處理
rl.on('line', function (line) {
  lines.push(line)
});

// 輸入結束，開始針對 lines 做處理
rl.on('close', function() {
  solve(lines)
})

// 上面都不用管，只需要完成這個 function 就好，可以透過 lines[i] 拿取內容
function solve(lines) {
  let arr=[]
  // arr[i]=i的個數
for(let i =1;i<=100;i++){
  arr[i]=0
}

for(let i =1;i<lines.length;i++){
  let n = Number(lines[i]);
  arr[n]++
}

for(let i=1;i<=100;i++){
  for(let j=1;j<=arr[i];j++){
    console.log(i)
  }
}
}

```

使用二分搜尋法，類似終極密碼，每次將數對切半比較

```js
var readline = require('readline');
var rl = readline.createInterface({
  input: process.stdin
});

var lines = []

// 讀取到一行，先把這一行加進去 lines 陣列，最後再一起處理
rl.on('line', function (line) {
  lines.push(line)
});

// 輸入結束，開始針對 lines 做處理
rl.on('close', function() {
  solve(lines)
})

// 上面都不用管，只需要完成這個 function 就好，可以透過 lines[i] 拿取內容
function solve(lines) {
  let [n, m] = lines[0].split(' ')
  let arr = []
  for(let i=0; i<n; i++) {
    arr[i] = Number(lines[i+1])
  }

  n = Number(n)

  for(let i=n+1; i<lines.length; i++) {
    let q = Number(lines[i])
    console.log(search(arr, q))
  }
}

function search(arr, q) {
  /*
    [............]
    ^L    ^M    ^R  

    [............]
    ^L   ^R 

  */
  let L = 0
  let R = arr.length - 1
  while(L <= R) {
    let M = Math.floor((L+R) / 2)
    if (arr[M] === q) {
      return M
    } else if (arr[M] > q) {
      R = M - 1
    } else {
      L = M + 1
    }
  }

  return -1
}

```

### [LIOJ | 陣列最短距離](https://oj.lidemy.com/problem/1049)

這題複習了一下`.map()`方法，可以遍歷陣列並帶入一個 function 

原本的解題沒有做判斷會導致超時，根據教學提示後加入判斷就可以成功 AC

```js
var readline = require('readline');
var rl = readline.createInterface({
  input: process.stdin
});

var lines = []

// 讀取到一行，先把這一行加進去 lines 陣列，最後再一起處理
rl.on('line', function (line) {
  lines.push(line)
});

// 輸入結束，開始針對 lines 做處理
rl.on('close', function() {
  solve(lines)
})

// 上面都不用管，只需要完成這個 function 就好，可以透過 lines[i] 拿取內容
function solve(lines) {
  let a = lines[1].split(' ').map(Number)
  let b = lines[2].split(' ').map(Number)
  let min = Infinity
  let r =0;
  for(let i =0;i<a.length;i++){
    for(let j=0;j<b.length;j++){
      let myy = Math.abs(Math.abs(a[i])-Math.abs(b[j]))		//判斷句離
      let myy2 = Math.abs(Math.abs(a[i])-Math.abs(b[j+1]))	//判斷下一個距離
      if(myy2>myy){										//判斷是否需要繼續跑 j 迴圈
        j=0
        i++
      }
      if(myy<min){
        min=myy;
      }
    }
  }
  console.log(min);
}


```

### [LIOJ | Two sum](https://oj.lidemy.com/problem/1050)

看了一下原來是 leetcode 的題目，原本還小緊張一下，結果發現原上 leetcode 上也有這麼簡單的題目

這邊有做一個判斷，只要**被加數>目標**那就可以跳過這個數，直接跑下一個迴圈

```js
var readline = require('readline');
var rl = readline.createInterface({
  input: process.stdin
});

var lines = []

// 讀取到一行，先把這一行加進去 lines 陣列，最後再一起處理
rl.on('line', function (line) {
  lines.push(line)
});

// 輸入結束，開始針對 lines 做處理
rl.on('close', function() {
  solve(lines)
})

// 上面都不用管，只需要完成這個 function 就好，可以透過 lines[i] 拿取內容
function solve(lines) {
  let n = lines[0].split(' ').map(Number)[1]
  let a = lines[1].split(' ').map(Number)
  let r = []
  let mr ='';
  for(let i=0;i<a.length;i++){
    for(let j=i+1;j<a.length;j++){
      if(j>n){
        i++
        j=0
      }
      if(a[i]+a[j]===n){
        r.push(i,j)
        return
      }
    }
  }
  r.sort((a,b)=>{
    return a-b
  })
  for(let i=0;i<r.length;i++){
    mr = mr + r[i] + ' '
  }
  console.log(mr)
}

```
