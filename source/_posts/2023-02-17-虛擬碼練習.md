### 字串反轉

給一個字串 str，請輸出 str 反過來的結果

範例輸入：hello

範例輸出：olleh

PS. 可以用 str[i] 取得第 i 個字，例如說 str="abc"，str[0] 就是 'a'

使用 JS 內建 `charAt(n)` 取得第 n 個字元，並使用字串相加輸入到 result 變數

```js
function str(str){
    let result  = '';
        for(let i=str.length-1 ;i>=0;i--){ 
         result+=str.charAt(i);
        }
    console.log(result);
}
str('hello');
```

```js
        for(let i=2;i<lines.length;i++){
```

1 *1 = 1 、2* 2 = 2、3 * 3 = 9 ...以此類推

第一種方法：將所有數開根號並相乘，看結果是否等同傳進來的數值，是的話回傳 true

```js
function solve(lines) {
    for(i=1;i<=100;i++){
        if(isSqtr(i)){
            console.log(i);
        }
    };
    function isSqtr(i){
        let r = Math.floor(Math.sqrt(i));
        return r * r === i ;
    }
}
```


### [LIOJ | 水仙花數](https://oj.lidemy.com/problem/1025)

解題思路：

1. 先將輸入分割為起始與終點
2. 將所有數字 for 遍歷並放入 function 判斷
3. function 中先將傳入的數字轉為字串再使用`.split('')` 將所有數字轉為字串的值分割
4. 設定變數`ff`為此陣列的 `.length` 則可以知道是幾位數的數字
5. 設定一個變數將所有分割的字串取 平方 ** ，並帶入`ff`，並將所有值相加
6. 判斷相加後的值與傳入的值是否相同

```js
var readline = require('readline');

var lines = []
var rl = readline.createInterface({
  input: process.stdin
});

rl.on('line', function (line) {
  lines.push(line)
});

rl.on('close', function() {
  solve(lines)
})

function solve(lines) {
    let N = lines[0].split(' ');
    let star = Number(N[0]); // 起始數
    let end = Number(N[1]); // 終點數
    for(let i=star;i<=end;i++){
        if(isflow(i)){
            console.log(i);
        }
    }
    function isflow(x){
        let arr =x.toString().split(''); //分割為陣列
        let ff = arr.length //幾位數
        let r = 0;
        for (let i =0 ; i<ff ; i++){
        r+=arr[i]**ff;
    }
    if(r==x){
        return true
    }
    return false
    }
}
```

```js
  solve(lines)
})
```

```js
    let n=Number(lines[0])
    let all = (lines.length-1)/4 //直接算出總共有幾組數(4個為1組)
    let now = 1; //用於儲存當前跑到的位數
for(let k=1;k<=all;k++){
    let arr =[];

        now++
    }
```
